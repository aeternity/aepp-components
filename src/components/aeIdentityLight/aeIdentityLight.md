Basic identity information
```js
const BN = require('bn.js')
const identity = {
  address: '0x35d8830ea35e6Df033eEdb6d5045334A4e34f9f9',
  name: 'My identity',
  balance: new BN('1337000000000000000', 10)
}

<ae-identity-light v-bind="identity" :collapsed="false" />
```
